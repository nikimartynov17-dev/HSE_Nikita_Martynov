# Задание 2.
# Пользователь вводит время в секундах. Программа переводит его в часы,
# минуты и секунды и выводит каждое значение отдельно.
# Используется приведение типов и метод .isdigit() для проверки ввода.
# =============================================================================

print("\n=== Задание 2: Перевод секунд в часы, минуты, секунды ===")

raw_input = input("Введите время в секундах (целое число): ")

# Проверяем, что введены только цифры
if raw_input.isdigit():
    total_seconds = int(raw_input)  # приведение типа: str -> int

    hours = total_seconds // 3600
    minutes = (total_seconds % 3600) // 60
    seconds = total_seconds % 60

    print(f"\nВведено секунд: {total_seconds}")
    print(f"Часы:   {hours}")
    print(f"Минуты: {minutes}")
    print(f"Секунды: {seconds}")
    print(f"Итого: {hours} ч. {minutes} мин. {seconds} сек.")
else:
    print("Ошибка: введены некорректные данные. Пожалуйста, введите целое положительное число.")


# =============================================================================
# Задание 3.
# Пользователь вводит цифру n от 1 до 9.
# Программа вычисляет сумму n + nn + nnn.
# Например, при n = 3: 3 + 33 + 333 = 369.
# =============================================================================

print("\n=== Задание 3: Сумма n + nn + nnn ===")

n_input = input("Введите цифру от 1 до 9: ")

if n_input.isdigit() and 1 <= int(n_input) <= 9:
    n = int(n_input)

    # Формируем числа nn и nnn из строки
    nn = int(str(n) * 2)    # например, "33"
    nnn = int(str(n) * 3)   # например, "333"

    result = n + nn + nnn

    print(f"\nn   = {n}")
    print(f"nn  = {nn}")
    print(f"nnn = {nnn}")
    print(f"\n{n} + {nn} + {nnn} = {result}")
else:
    print("Ошибка: введите целое число от 1 до 9.")
